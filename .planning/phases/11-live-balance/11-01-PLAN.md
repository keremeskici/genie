          args: [wallet as `0x${string}`],

        return c.json({ balance, currency: 'USDC' });
      } catch (err) {

        return c.json({ error: 'FETCH_FAILED', message: 'Could not retrieve balance' }, 500);
      }

    **Create `apps/web/src/hooks/useBalance.ts`** (per D-03, D-04, D-05, D-06, D-07):

    ```typescript
    import { useState, useEffect, useCallback } from 'react';

    const API_URL = process.env.NEXT_PUBLIC_API_URL ?? '';

    export function useBalance(walletAddress: string) {
      const [balance, setBalance] = useState<string | null>(null);
      const [loading, setLoading] = useState(false);
      const [error, setError] = useState(false);

      const fetchBalance = useCallback(async () => {
        if (!walletAddress) return;
        setLoading(true);
        setError(false);
        try {
          const res = await fetch(`${API_URL}/api/balance?wallet=${walletAddress}`);
          if (!res.ok) throw new Error(`HTTP ${res.status}`);
          const data = await res.json();
          const parsed = parseFloat(data.balance);
          if (isNaN(parsed)) throw new Error('invalid balance');
          setBalance(parsed.toFixed(2));
        } catch (err) {
          console.error('[useBalance] fetch failed:', err);
          setError(true);
        } finally {
          setLoading(false);
        }
      }, [walletAddress]);

      useEffect(() => {
        fetchBalance();
      }, [fetchBalance]);

      return { balance, loading, error, refetch: fetchBalance };
    }
    ```

    Key details:
    - URL is `${API_URL}/api/balance` (NEXT_PUBLIC_API_URL does NOT include /api — verified from onboarding/page.tsx which uses `${API_URL}/api/users/profile`)
    - `fetchBalance` is memoized with `useCallback` and exposed as `refetch` for D-04
    - Guard: early return if `!walletAddress` prevents empty-param fetch during SSR hydration
    - `parseFloat(data.balance).toFixed(2)` per D-07 ($X.XX format)
    - No thousand separators per D-08
