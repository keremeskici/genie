**Middleware (per D-04, D-05, D-06):**

Replace the entire `apps/web/middleware.ts` with:

```typescript
import { auth } from '@/auth';
import { NextResponse } from 'next/server';

export default auth((req) => {
  const { pathname } = req.nextUrl;

  const isPublicPath =
    pathname === '/' ||
    pathname.startsWith('/api/auth') ||
    pathname.startsWith('/api/verify-proof') ||
    pathname.startsWith('/api/rp-signature') ||
    pathname.startsWith('/_next/static') ||
    pathname.startsWith('/_next/image') ||
    pathname === '/favicon.ico';

  if (!req.auth && !isPublicPath) {
    return NextResponse.redirect(new URL('/', req.url));
  }
});

export const config = {
  matcher: ['/((?!_next/static|_next/image|favicon.ico).*)'],
};
```

Key points:
- Uses `auth(callback)` form, NOT `export { auth as middleware }` (Pitfall 1 from research)
- `req.auth` is the session object in callback form; falsy when no session
- Public paths list per D-05: `/`, `/api/auth/*`, `/api/verify-proof`, `/api/rp-signature`, static assets, favicon
- Protected layout at `(protected)/layout.tsx` is kept as backup per D-06 — do NOT modify it
